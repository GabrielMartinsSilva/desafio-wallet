200 def _parse_config(self, config_parser, parent=None): 

201 config_dict = {} 

202 for option_name in config_parser.options(self.program_name): 

203 if option_name not in self.config_options: 

204 LOG.debug( 

205 'Option "%s" is not registered. Ignoring.', option_name 

206 ) 

207 continue 

208 option = self.config_options[option_name] 

209 

210 # Use the appropriate method to parse the config value 

211 method = config_parser.get 

212 if option.type is int or option.action == "count": 

213 method = config_parser.getint 

214 elif option.action in self.GETBOOL_ACTIONS: 

215 method = config_parser.getboolean 

216 

217 value = method(self.program_name, option_name) 

218 LOG.debug('Option "%s" returned value: %r', option_name, value) 

219 

220 final_value = self._normalize_value(option, value, parent) 

221 config_dict[option.config_name] = final_value 

222 

223 return config_dict
