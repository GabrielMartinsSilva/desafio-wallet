97 def parse_preliminary_options(self, argv): 

98 # type: (List[str]) -> Tuple[argparse.Namespace, List[str]] 

99 """Get preliminary options from the CLI, pre-plugin-loading. 

100 

101 We need to know the values of a few standard options so that we can 

102 locate configuration files and configure logging. 

103 

104 Since plugins aren't loaded yet, there may be some as-yet-unknown 

105 options; we ignore those for now, they'll be parsed later when we do 

106 real option parsing.

108 :param list argv: 

109 Command-line arguments passed in directly. 

110 :returns: 

111 Populated namespace and list of remaining argument strings. 

112 :rtype: 

113 (argparse.Namespace, list) 

114 """ 

115 args, rest = self.prelim_arg_parser.parse_known_args(argv) 

116 # XXX (ericvw): Special case "forwarding" the output file option so

120 return args, rest

275 if not self.options.benchmark:
